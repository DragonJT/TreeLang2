
void DrawEquilateralTriangle(int x, int y, Color color)
{
    var a = Vector2(x-100, y+100);
    var b = Vector2(x, y-100);
    var c = Vector2(x+100, y+100);
    DrawTriangle(a,c,b,color);
}

void Main()
{
    InitWindow(800, 600, "Test");
    SetTargetFPS(120);
    var y = 20;
    while !WindowShouldClose() 
    {
        BeginDrawing();
        ClearBackground(BLUE);
        DrawEquilateralTriangle(200, y, RED);
        DrawEquilateralTriangle(400, y, GREEN);
        EndDrawing();
        y = y + 1;
        if y > 450
        {
            y = 20;
        }
    }
}
